📦Installation

From npm

# Install the WASM package npm install wasm-spatial-core

From Source

git clone https://github.com/reed-soul/wasm-spatial-core.git cd wasm-spatial-core wasm-pack build --target web --release --out-dir pkg -- --features point-cloud,geotiff

🚀Quick Start

import init, {
  parsePointCloudAuto,
  buildOctree,
  generateTileset,
} from 'wasm-spatial-core';

await init();

// Parse any point cloud format (LAS, LAZ, COPC, PLY, OBJ...)
const cloud = parsePointCloudAuto(lasBytes);

// Build octree for spatial partitioning
const octree = buildOctree(cloud.positions());

// Generate 3D Tiles tileset (pnts + tileset.json)
const tiles = generateTileset(cloud.positions(), 50000, 21);
console.log(tiles.tilesetJson());
console.log(`Tiles: ${tiles.tileCount()}`);
